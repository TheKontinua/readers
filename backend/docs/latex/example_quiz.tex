\begin{mdframed}[frametitle={Ideal Gas Law}]

$$PV = nRT$$

where:
\begin{itemize}
\item[] $P =$ pressure in pascals
\item[] $V =$ volume in cubic meters
\item[] $n =$ number of molecules in moles
\item[] $R =$ the molar gas constant: 8.31446
\item[] $T =$ temperature in Kelvin
\end{itemize}
\end{mdframed}

$$\mu=E(x)=np$$
$$\sigma^2_x=np(1-p)$$
